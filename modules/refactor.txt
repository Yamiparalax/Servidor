# 5. DESTINATÁRIOS VIA BIGQUERY

**JAMAIS** deixe listas de e-mails hardcoded. Use esta função para buscar da tabela `Registro_automacoes`:

```python
def obter_destinatarios(logger):
    """Busca destinatários no BigQuery."""
    try:
        import pandas_gbq
        sql = f"""
            SELECT emails_principais, emails_cc 
            FROM `datalab-pagamentos.ADMINISTRACAO_CELULA_PYTHON.Registro_automacoes` 
            WHERE lower(TRIM(metodo_automacao)) = lower(TRIM('{Path(__file__).stem}'))
            LIMIT 1
        """
        df = pandas_gbq.read_gbq(sql, project_id="datalab-pagamentos")
        if df.empty: return [], []
        
        def limpar(raw):
            if not raw or str(raw).lower() == 'nan': return []
            return [x.strip() for x in str(raw).replace(';', ',').split(',') if '@' in x]
            
        return limpar(df.iloc[0]['emails_principais']), limpar(df.iloc[0]['emails_cc'])
    except Exception as e:
        logger.error(f"Erro ao buscar destinatários: {e}")
        return [], []
```
