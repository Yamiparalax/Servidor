# 5. DESTINATÁRIOS VIA BIGQUERY

**JAMAIS** deixe listas de e-mails hardcoded. Use esta função para buscar da tabela `Registro_automacoes`:

```python
def obter_destinatarios(logger):
    """Busca destinatários no BigQuery."""
    try:
        import pandas_gbq
        sql = f"""
            SELECT emails_principais, emails_cc 
            FROM `datalab-pagamentos.ADMINISTRACAO_CELULA_PYTHON.Registro_automacoes` 
            WHERE lower(TRIM(metodo_automacao)) = lower(TRIM('{Path(__file__).stem}'))
            LIMIT 1
        """
        df = pandas_gbq.read_gbq(sql, project_id="datalab-pagamentos")
        if df.empty: return [], []
        
        def limpar(raw):
            if not raw or str(raw).lower() == 'nan': return []
            return [x.strip() for x in str(raw).replace(';', ',').split(',') if '@' in x]
            
        return limpar(df.iloc[0]['emails_principais']), limpar(df.iloc[0]['emails_cc'])
    except Exception as e:
        logger.error(f"Erro ao buscar destinatários: {e}")
        return [], []
```

## 7.1. INSERÇÃO SEGURA (DEDUPLICAÇÃO VIA STAGING) - OBRIGATÓRIO

Para evitar duplicatas, **NUNCA** faça append direto na tabela final. Use sempre o padrão **Staging Table**:

1. Crie o nome da tabela temporária: `TABELA_STAGING = f"{TABELA_FINAL}_staging"`.
2. Suba o DataFrame para a Staging (**Overwrite**):
    ```python
    pandas_gbq.to_gbq(df, TABELA_STAGING, if_exists='replace', ...)
    ```
3. Execute SQL para inserir **apenas linhas novas** na final (Comparação Total):
    ```python
    sql_dedup = f"""
        INSERT INTO `{TABELA_FINAL}`
        SELECT * FROM `{TABELA_STAGING}` S
        WHERE NOT EXISTS (
            SELECT 1 FROM `{TABELA_FINAL}` F
            WHERE TO_JSON_STRING(S) = TO_JSON_STRING(F)
        )
    """
    # Ou explicitamente comparando colunas se TO_JSON_STRING for pesado, mas TO_JSON é o padrão universal pedido.
    ```
    (Nota: Se preferir comparar colunas explicitamente `S.col1 = F.col1 AND ...`, pode, mas deve cobrir TODAS as colunas).

4. Delete a Staging:
    ```python
    client.query(f"DROP TABLE IF EXISTS `{TABELA_STAGING}`")
    ```

```python
def _exibir_dialogo_inicial():
    """Exibe diálogo PyQt5 para configurar execução manual."""
    try:
        from PyQt5.QtWidgets import QApplication, QDialog, QVBoxLayout, QLabel, QComboBox, QLineEdit, QPushButton, QMessageBox
        from PyQt5.QtCore import Qt
    except ImportError:
        print("PyQt5 não encontrado. Rodando direto.")
        return "AUTO", f"{getpass.getuser()}@c6bank.com"

    app = QApplication.instance() or QApplication([])
    dialog = QDialog()
    dialog.setWindowTitle(f"Config: {NOME_SCRIPT}")
    dialog.resize(300, 300)
    
    layout = QVBoxLayout(dialog)
    
    # Modo
    layout.addWidget(QLabel("Modo de Execução:"))
    cmb_modo = QComboBox()
    cmb_modo.addItems(["AUTO", "SOLICITACAO"])
    layout.addWidget(cmb_modo)
    
    # Usuário
    layout.addWidget(QLabel("Usuário:"))
    txt_user = QLineEdit(f"{getpass.getuser()}@c6bank.com")
    layout.addWidget(txt_user)
    
    # Botão
    btn_run = QPushButton("EXECUTAR")
    btn_run.setFixedHeight(40)
    # Estilo 'Netflix' básico
    btn_run.setStyleSheet("background-color: #E50914; color: white; font-weight: bold; border-radius: 4px;")
    
    def on_click():
        if not txt_user.text().strip():
             QMessageBox.warning(dialog, "Erro", "Usuário obrigatório.")
             return
        dialog.accept()
        
    btn_run.clicked.connect(on_click)
    layout.addWidget(btn_run)
    layout.addStretch()
    
    if dialog.exec_():
        return cmb_modo.currentText(), txt_user.text().strip()
    return None, None

if __name__ == "__main__":
    # Se rodado manualmente, pergunta configurações
    modo, user = _exibir_dialogo_inicial()
    if modo and user:
        os.environ["MODO_EXECUCAO"] = modo
        os.environ["USUARIO_EXEC"] = user
        os.environ["OBSERVACAO"] = "Execução Manual via GUI"
        main()
    else:
        print("Execução cancelada pelo usuário.")
```
